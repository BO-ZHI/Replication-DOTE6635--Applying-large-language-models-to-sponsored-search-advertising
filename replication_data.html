# =============================================
# Table 3 IT  Campaign Performance Metrics
# =============================================
# For each experimental group (Human Novice / GPT-3 / PPLM),
# compute aggregate campaign KPIs and run goodness-of-fit or
# non-parametric tests to check whether groups differ.

# 1) Number of ads created per experimental group
No_ads = IT_tab3.groupby('Experimental.group')['Impr.'].count().reset_index()
No_ads.columns = ['Category', 'No_ads']
display_summary("Number of Ads per Group (IT)", No_ads)

# 2) Total impressions per group + chi-squared goodness-of-fit
#    (H0: impressions are uniformly distributed across groups)
No_impr = IT_tab3.groupby('Experimental.group')['Impr.'].sum().reset_index()
No_impr.columns = ['Category', 'Impressions']
display_summary("Impressions per Group (IT)", No_impr)
chi2_impr = stats.chisquare(No_impr['Impressions'])
print(f"Chi-squared test: statistic={chi2_impr.statistic:.4f}, p-value={chi2_impr.pvalue:.4f}")

# 3) Total clicks per group + chi-squared
No_clicks = IT_tab3.groupby('Experimental.group')['Klicks'].sum().reset_index()
No_clicks.columns = ['Category', 'Clicks']
display_summary("Clicks per Group (IT)", No_clicks)
chi2_clicks = stats.chisquare(No_clicks['Clicks'])
print(f"Chi-squared test: statistic={chi2_clicks.statistic:.4f}, p-value={chi2_clicks.pvalue:.4f}")

# 4) Estimated conversions = clicks  group-specific conversion rate
#    Conversion rates come from Table A19 in the Web Appendix
conv_rates = [0.0197, 0.02, 0.02]
No_Conversions = np.round(No_clicks['Clicks'].values * conv_rates).astype(int)
display_summary("Conversions per Group (IT)",
                pd.DataFrame({'Category': No_clicks['Category'], 'Conversions': No_Conversions}))
chi2_conv = stats.chisquare(No_Conversions)
print(f"Chi-squared test: statistic={chi2_conv.statistic:.4f}, p-value={chi2_conv.pvalue:.4f}")

# 5) Total advertising costs per group + chi-squared
Costs = IT_tab3.groupby('Experimental.group')['Kosten'].sum().reset_index()
Costs.columns = ['Category', 'Total_Cost']
display_summary("Costs per Group (IT)", Costs)
chi2_costs = stats.chisquare(Costs['Total_Cost'])
print(f"Chi-squared test: statistic={chi2_costs.statistic:.4f}, p-value={chi2_costs.pvalue:.4f}")

# 6) Cost per Click (CPC)  descriptive statistics (median, IQR, range)
CPC = IT_tab3.groupby('Experimental.group')['Durchschn..CPC'].agg(
    Median='median', IQR=lambda x: stats.iqr(x, nan_policy='omit'),
    Min='min', Max='max'
).reset_index()
CPC.columns = ['Category', 'Median', 'IQR', 'Min', 'Max']
display_summary("CPC Statistics per Group (IT)", CPC)

# Kruskal-Wallis H-test (non-parametric ANOVA) for CPC across groups
groups_cpc = [g['Durchschn..CPC'].dropna().values for _, g in IT_tab3.groupby('Experimental.group')]
kw_cpc = stats.kruskal(*groups_cpc)
print(f"Kruskal-Wallis test (CPC): statistic={kw_cpc.statistic:.4f}, p-value={kw_cpc.pvalue:.4f}")
